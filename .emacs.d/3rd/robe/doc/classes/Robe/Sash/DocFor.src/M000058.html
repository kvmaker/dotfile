      def visibility
        owner, name = @method.owner, @method.name
        if owner.public_instance_methods(false).include?(name)
          :public
        elsif owner.protected_instance_methods(false).include?(name)
          :protected
        elsif owner.private_instance_methods(false).include?(name)
          :private
        end
      end
