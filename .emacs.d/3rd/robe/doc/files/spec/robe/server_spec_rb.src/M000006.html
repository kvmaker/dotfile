  def start_and_send(handler, request)
    $port += 1
    server = described_class.new(handler, $port)

    Thread.new do
      Thread.current.abort_on_exception = true
      server.start
    end

    server.wait_for_it

    http = Net::HTTP.new("127.0.0.1", $port)
    http.request(request)
  ensure
    server.shutdown
  end
